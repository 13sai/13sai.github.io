小型商城商品模块设计

最近有做到类商城小型项目，主要负责商品模块，便分享下商品模块设计思路。

主要分为5个表：


商品作为主表，与规格表、属性表、sku表直接关联。


ER关系图如下：


SQL语句如下：

-- ----------------------------
-- Table structure for goods
-- ----------------------------
DROP TABLE IF EXISTS `goods`;
CREATE TABLE `goods`  (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `mid` int(11) NOT NULL DEFAULT 0 COMMENT '店铺id',
  `goods_sn` varchar(128) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL COMMENT '商品编码',
  `title` varchar(128) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `main_pic` varchar(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '商品主图',
  `pics` varchar(1024) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '图集',
  `cate_id` int(11) NOT NULL DEFAULT 0 COMMENT '分类id',
  `outer_price` decimal(10, 2) NOT NULL DEFAULT 0.00 COMMENT '外部价格 单位：元',
  `inner_price` decimal(10, 2) NOT NULL DEFAULT 0.00 COMMENT '价格 单位：元',
  `description` text CHARACTER SET utf8 COLLATE utf8_general_ci NULL,
  `is_delete` tinyint(1) NOT NULL DEFAULT 0 COMMENT '0 正常 1 删除',
  `status` tinyint(2) NOT NULL DEFAULT 4 COMMENT '2审核，3上架，4店铺下架，5平台下架，6拒绝上架',
  `create_at` timestamp(0) NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '添加时间',
  `update_at` timestamp(0) NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP(0) COMMENT '更新时间',
  `stock_num` int(11) NOT NULL DEFAULT 0 COMMENT '库存',
  `sale_num` int(11) NOT NULL DEFAULT 0 COMMENT '销量',
  `sku_type` tinyint(1) NOT NULL DEFAULT 2 COMMENT '1: 单规格 2：多规格',
  `item_url` varchar(512) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL DEFAULT '' COMMENT '商品链接',
  `sell_out` tinyint(1) NOT NULL DEFAULT 0 COMMENT '已卖光',
  PRIMARY KEY (`id`) USING BTREE,
  INDEX `shop_id`(`mid`) USING BTREE,
  INDEX `cate_id`(`cate_id`) USING BTREE,
  CONSTRAINT `cate_id` FOREIGN KEY (`cate_id`) REFERENCES `goods_category` (`id`) ON DELETE RESTRICT ON UPDATE RESTRICT
) ENGINE = InnoDB AUTO_INCREMENT = 1 CHARACTER SET = utf8 COLLATE = utf8_general_ci COMMENT = '商品表' ROW_FORMAT = Dynamic;


– —————————-
– Table structure for goods_attr_key
– —————————-
DROP TABLE IF EXISTS goods_attr_key;
CREATE TABLE goods_attr_key  (
  id int(11) NOT NULL AUTO_INCREMENT,
  goods_id int(11) NOT NULL DEFAULT 0 COMMENT '商品id',
  attr varchar(50) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  PRIMARY KEY (id) USING BTREE,
  INDEX goods_id(goods_id) USING BTREE,
  CONSTRAINT goods_id FOREIGN KEY (goods_id) REFERENCES goods (id) ON DELETE RESTRICT ON UPDATE RESTRICT
) ENGINE = InnoDB AUTO_INCREMENT = 1 CHARACTER SET = utf8 COLLATE = utf8_general_ci COMMENT = '商品规格' ROW_FORMAT = Dynamic;


– —————————-
– Table structure for goods_attr_value
– —————————-
DROP TABLE IF EXISTS goods_attr_value;
CREATE TABLE goods_attr_value  (
  id int(11) NOT NULL AUTO_INCREMENT,
  attr_id int(11) NOT NULL COMMENT '属性项目id',
  attr_name varchar(50) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  goods_id int(11) NOT NULL DEFAULT 0 COMMENT '商品id',
  PRIMARY KEY (id) USING BTREE,
  INDEX goods_id(goods_id) USING BTREE,
  INDEX attr_id(attr_id) USING BTREE,
  CONSTRAINT val_goods_id FOREIGN KEY (goods_id) REFERENCES goods (id) ON DELETE RESTRICT ON UPDATE RESTRICT,
  CONSTRAINT attr_id FOREIGN KEY (attr_id) REFERENCES goods_attr_key (id) ON DELETE RESTRICT ON UPDATE RESTRICT
) ENGINE = InnoDB AUTO_INCREMENT = 1 CHARACTER SET = utf8 COLLATE = utf8_general_ci COMMENT = '商品属性' ROW_FORMAT = Dynamic;


– —————————-
– Table structure for goods_category
– —————————-
DROP TABLE IF EXISTS goods_category;
CREATE TABLE goods_category  (
  id int(11) NOT NULL AUTO_INCREMENT,
  parent_id int(11) NOT NULL DEFAULT 0 COMMENT '父级id',
  cate_name varchar(128) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '分类名称',
  sort tinyint(4) NOT NULL DEFAULT 0 COMMENT '排序',
  create_at timestamp(0) NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (id) USING BTREE
) ENGINE = InnoDB AUTO_INCREMENT = 1 CHARACTER SET = utf8 COLLATE = utf8_general_ci COMMENT = '商品分类' ROW_FORMAT = Dynamic;


– —————————-
– Table structure for goods_sku
– —————————-
DROP TABLE IF EXISTS goods_sku;
CREATE TABLE goods_sku  (
  id int(11) NOT NULL AUTO_INCREMENT,
  goods_id int(11) NOT NULL DEFAULT 0 COMMENT '商品id',
  mid int(11) NOT NULL DEFAULT 0 COMMENT '店铺id',
  attr_ids varchar(512) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL DEFAULT '' COMMENT '规格集合',
  attr_names varchar(512) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL DEFAULT '' COMMENT '属性集合',
  goods_sku_sn varchar(64) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '商品sku编码',
  outer_price decimal(10, 2) NOT NULL DEFAULT 0.00 COMMENT '外部价格 单位：元',
  inner_price decimal(10, 2) NOT NULL DEFAULT 0.00 COMMENT '价格 单位：元',
  stock_num int(11) NOT NULL DEFAULT 0 COMMENT '可用库存',
  dj_stock_num int(11) NOT NULL DEFAULT 0 COMMENT '冻结库存',
  sale_num int(11) NOT NULL DEFAULT 0 COMMENT '销量',
  create_at timestamp(0) NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '添加时间',
  update_at timestamp(0) NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP(0) COMMENT '更新时间',
  PRIMARY KEY (id) USING BTREE,
  INDEX goods_id(goods_id) USING BTREE,
  CONSTRAINT goods_id_sku FOREIGN KEY (goods_id) REFERENCES goods (id) ON DELETE RESTRICT ON UPDATE RESTRICT
) ENGINE = InnoDB AUTO_INCREMENT = 1 CHARACTER SET = utf8 COLLATE = utf8_general_ci COMMENT = '商品sku' ROW_FORMAT = Dynamic;
